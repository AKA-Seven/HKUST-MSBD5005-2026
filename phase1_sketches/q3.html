import pandas as pd import networkx as nx import matplotlib.pyplot as plt import matplotlib.patches as mpatches from pathlib import Path # ================= Config ================= PROJECT_ROOT = Path(__file__).resolve().parents[2] OUTPUT_DIR = PROJECT_ROOT / "outputs" / "q3" anomaly_path = OUTPUT_DIR / "anomaly_delta.csv" bridge_path = OUTPUT_DIR / "bridge_companies.csv" relay_path = OUTPUT_DIR / "relay_chains.csv" def is_valid_name(name): if pd.isna(name): return False s = str(name).strip() return s != '' and s != '-49' and not s.isdigit() def load_data(): score_col = 'suspicious_revival_score' if 'suspicious_revival_score' in anomaly_df.columns else 'score' anomaly_df = pd.read_csv(anomaly_path) # high-risk companies (score > 50) -> highlighted actors high_df = anomaly_df[anomaly_df[score_col] > 50] if high_df.empty: high_df = anomaly_df.nlargest(3, score_col) highlighted = [c for c in high_df['company'].tolist() if is_valid_name(c)] if not highlighted: highlighted = [c for c in anomaly_df['company'].tolist() if is_valid_name(c)][:1] # bridge companies (top 1) bridge_df = pd.read_csv(bridge_path) bridge_df = bridge_df[bridge_df['company'].apply(is_valid_name)] top_bridges = bridge_df.nlargest(1, 'bridge_scope')['company'].tolist() # relay chains (top 1) relay_df = pd.read_csv(relay_path) relay_df = relay_df[relay_df['predecessor'].apply(is_valid_name) & relay_df['successor'].apply(is_valid_name)] top_relays = relay_df.nlargest(1, 'shared_partner_count')[['predecessor', 'successor']].values.tolist() # normal companies (low score) normal_df = anomaly_df[anomaly_df[score_col] < 10] normal_df = normal_df[normal_df['company'].apply(is_valid_name)] if len(normal_df) >= 2: normal = normal_df.head(3)['company'].tolist() else: normal = ["Serengeti Inc. Ltd Seabed", "Costa de la Felicidad Shipping"] return highlighted, top_bridges, top_relays, normal def build_original_edges(highlighted, top_bridges, top_relays, normal): edges = [] if len(normal) >= 2: edges.append((normal[0], normal[1])) if len(normal) >= 3: edges.append((normal[1], normal[2])) if highlighted: edges.append((highlighted[0], highlighted[0])) # self-loop for shell return edges def build_trusted_edges(highlighted, top_bridges, top_relays, normal): bridge_edges = [] relay_edges = [] burst_edges = [] if top_bridges and normal: b = top_bridges[0] for n in normal[:2]: bridge_edges.append((b, n)) if top_relays: pred, succ = top_relays[0] relay_edges.append((pred, succ)) if highlighted and normal: sh = highlighted[0] for n in normal: burst_edges.append((sh, n)) # merge all trusted added edges into one list (same color #196d69) trusted = bridge_edges + relay_edges + burst_edges return trusted def plot_comparison(original_edges, trusted_edges, highlighted, top_bridges, top_relays, normal): # Collect all nodes all_nodes = set() for u, v in original_edges: all_nodes.add(u); all_nodes.add(v) for u, v in trusted_edges: all_nodes.add(u); all_nodes.add(v) # Node roles: highlighted if in high-risk OR bridge OR relay entity highlighted_set = set(highlighted) | set(top_bridges) for pred, succ in top_relays: highlighted_set.add(pred); highlighted_set.add(succ) node_color = {} for n in all_nodes: if n in highlighted_set: node_color[n] = '#ad5c16' # highlighted actor else: node_color[n] = '#7f8c8d' # normal (gray-blue) # Layout using all edges G_full = nx.Graph() G_full.add_edges_from(original_edges + trusted_edges) pos = nx.spring_layout(G_full, seed=42, k=1.5, iterations=60) fig, (ax1, ax2) = plt.subplots(1, 2, figsize=(18, 9), facecolor='#f5efe2') fig.suptitle('Before / After Network Comparison\nQ3 Graph Completion Analysis', fontsize=16, fontweight='bold', y=0.98, color='#14213d') # ---------- Before ---------- G_before = nx.Graph() G_before.add_edges_from(original_edges) node_colors_before = [node_color[n] for n in G_before.nodes()] node_sizes_before = [1600 if n in highlighted_set else 1100 for n in G_before.nodes()] nx.draw_networkx_nodes(G_before, pos, ax=ax1, node_color=node_colors_before, node_size=node_sizes_before, edgecolors='black', linewidths=1.2) nx.draw_networkx_edges(G_before, pos, ax=ax1, edge_color='#203553', width=2.0, style='dashed', alpha=0.7) # Labels for node, (x, y) in pos.items(): if node in G_before.nodes(): ax1.text(x, y, node, fontsize=8, fontweight='bold', ha='center', va='center', bbox=dict(boxstyle="round,pad=0.2", facecolor="white", edgecolor="gray", alpha=0.85)) # Annotations (English) if top_bridges: bnode = top_bridges[0] if bnode in pos: ax1.annotate("Isolated bridge", xy=pos[bnode], xytext=(-60, -25), textcoords='offset points', ha='center', fontsize=9, color='#ad5c16', arrowprops=dict(arrowstyle='->', color='#ad5c16', lw=1.2), bbox=dict(boxstyle="round,pad=0.2", facecolor="white", alpha=0.7)) if top_relays: pred, succ = top_relays[0] if pred in pos and succ in pos: mx = (pos[pred][0] + pos[succ][0])/2 my = (pos[pred][1] + pos[succ][1])/2 ax1.annotate("Broken chain\n(inactive)", xy=(mx, my), xytext=(0, 35), textcoords='offset points', ha='center', fontsize=9, color='#203553', arrowprops=dict(arrowstyle='->', color='#203553', lw=1.2), bbox=dict(boxstyle="round,pad=0.2", facecolor="white", alpha=0.7)) if highlighted: sh = highlighted[0] if sh in pos: ax1.annotate("Shell self-loop", xy=pos[sh], xytext=(50, -40), textcoords='offset points', ha='center', fontsize=9, color='#ad5c16', arrowprops=dict(arrowstyle='->', color='#ad5c16', lw=1.2), bbox=dict(boxstyle="round,pad=0.2", facecolor="white", alpha=0.7)) ax1.set_title("Before: Original Graph", fontsize=13, fontweight='bold', pad=15, color='#14213d') ax1.set_facecolor('#f9f9f9') for spine in ax1.spines.values(): spine.set_visible(False) ax1.set_xticks([]); ax1.set_yticks([]) ax1.set_xlim(-1.5, 1.5); ax1.set_ylim(-1.2, 1.2) # ---------- After ---------- G_after = nx.Graph() G_after.add_edges_from(original_edges + trusted_edges) node_colors_after = [node_color[n] for n in G_after.nodes()] node_sizes_after = [1600 if n in highlighted_set else 1100 for n in G_after.nodes()] nx.draw_networkx_nodes(G_after, pos, ax=ax2, node_color=node_colors_after, node_size=node_sizes_after, edgecolors='black', linewidths=1.2) # Original edges (dashed, #203553) if original_edges: nx.draw_networkx_edges(G_after, pos, ax=ax2, edgelist=original_edges, edge_color='#203553', width=2.0, style='dashed', alpha=0.6) # Trusted added edges (solid, #196d69) if trusted_edges: nx.draw_networkx_edges(G_after, pos, ax=ax2, edgelist=trusted_edges, edge_color='#196d69', width=3.5, style='solid', alpha=0.9) for node, (x, y) in pos.items(): if node in G_after.nodes(): ax2.text(x, y, node, fontsize=8, fontweight='bold', ha='center', va='center', bbox=dict(boxstyle="round,pad=0.2", facecolor="white", edgecolor="gray", alpha=0.85)) # Optional small label on a new edge if trusted_edges: u, v = trusted_edges[0] if u in pos and v in pos: mx = (pos[u][0] + pos[v][0])/2 my = (pos[u][1] + pos[v][1])/2 ax2.annotate("new link", xy=(mx, my), xytext=(10, 10), textcoords='offset points', fontsize=8, color='#196d69', ha='center', bbox=dict(boxstyle="round,pad=0.1", facecolor="white", alpha=0.6)) ax2.set_title("After: Completed Graph", fontsize=13, fontweight='bold', pad=15, color='#14213d') ax2.set_facecolor('#f9f9f9') for spine in ax2.spines.values(): spine.set_visible(False) ax2.set_xticks([]); ax2.set_yticks([]) ax2.set_xlim(-1.5, 1.5); ax2.set_ylim(-1.2, 1.2) # Legend (English, using requested colors) legend_elements = [ mpatches.Patch(facecolor='#7f8c8d', edgecolor='black', label='Normal company'), mpatches.Patch(facecolor='#ad5c16', edgecolor='black', label='Highlighted actor'), plt.Line2D([0], [0], color='#203553', lw=2, linestyle='dashed', label='original structure'), plt.Line2D([0], [0], color='#196d69', lw=3, label='trusted added edge') ] fig.legend(handles=legend_elements, loc='lower center', ncol=4, fontsize=10, bbox_to_anchor=(0.5, -0.02), frameon=False, facecolor='#f5efe2', edgecolor='none') plt.tight_layout(rect=[0, 0.03, 1, 0.95]) return fig def main(): highlighted, top_bridges, top_relays, normal = load_data() print("Highlighted actors:", highlighted[:1]) print("Bridge company:", top_bridges[:1]) print("Relay chain:", top_relays[:1]) print("Normal companies:", normal) original = build_original_edges(highlighted, top_bridges, top_relays, normal) trusted = build_trusted_edges(highlighted, top_bridges, top_relays, normal) fig = plot_comparison(original, trusted, highlighted, top_bridges, top_relays, normal) fig.savefig('Q3_before_after_network.png', dpi=300, bbox_inches='tight', facecolor='#f5efe2') plt.show() if __name__ == "__main__": main()
